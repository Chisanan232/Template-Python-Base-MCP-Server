# Examples and Tutorials

This section provides practical examples demonstrating how to use the MCP server template for common use cases.

## Example 1: Simple Calculator Tool

A basic example showing how to create a simple calculator tool.

### Step 1: Define Models

**Input Model** (`src/mcp/models/input/calculator_input.py`):

```python
from pydantic import BaseModel, Field

class CalculatorInput(BaseModel):
    """Input for calculator tool."""
    
    expression: str = Field(
        description="Mathematical expression to evaluate",
        examples=["2 + 2", "sqrt(16)", "3 * 4"]
    )
    precision: int = Field(
        default=2,
        ge=0,
        le=10,
        description="Decimal precision for results"
    )
```

**Output Model** (`src/mcp/models/output/calculator_output.py`):

```python
from pydantic import BaseModel, Field
from typing import Optional

class CalculatorOutput(BaseModel):
    """Output from calculator tool."""
    
    result: float = Field(description="Calculation result")
    expression: str = Field(description="Original expression")
    error: Optional[str] = Field(
        default=None,
        description="Error message if calculation failed"
    )
```

### Step 2: Implement Tool

**Tool Implementation** (`src/mcp/tools/calculator.py`):

```python
import math
from src.mcp import mcp_factory
from src.mcp.models.input.calculator_input import CalculatorInput
from src.mcp.models.output.calculator_output import CalculatorOutput

@mcp_factory.get().tool()
def calculate(input: CalculatorInput) -> CalculatorOutput:
    """
    Perform mathematical calculations.
    
    Supports basic arithmetic and math functions.
    """
    try:
        # Create safe namespace for eval
        safe_dict = {
            'sqrt': math.sqrt,
            'sin': math.sin,
            'cos': math.cos,
            'tan': math.tan,
            'pi': math.pi,
            'e': math.e,
        }
        
        result = eval(input.expression, {"__builtins__": {}}, safe_dict)
        
        return CalculatorOutput(
            result=round(float(result), input.precision),
            expression=input.expression
        )
    except Exception as e:
        return CalculatorOutput(
            result=0,
            expression=input.expression,
            error=f"Calculation error: {str(e)}"
        )
```

### Step 3: Test the Tool

**Test File** (`test/unit_test/mcp/test_calculator.py`):

```python
import pytest

    """Test math functions."""

    output = calculate(input_data)
    
    assert output.result == 4.0
    assert output.error is None

def test_invalid_expression():
    """Test error handling."""
    input_data = CalculatorInput(expression="invalid expression")
    output = calculate(input_data)
    
    assert output.result == 0
    assert output.error is not None
```

## Example 2: Text Processing API

An example showing how to create a web API endpoint for text processing.

### Step 1: Define Models

**Request Model** (`src/web_server/models/request/text_request.py`):

```python
from pydantic import BaseModel, Field

class TextProcessRequest(BaseModel):
    """Request for text processing."""
    
    text: str = Field(
        description="Text to process",
        min_length=1,
        max_length=10000
    )
    operation: str = Field(
        description="Operation to perform",
        pattern="^(uppercase|lowercase|reverse|word_count)$"
    )
```

**Response Model** (`src/web_server/models/response/text_response.py`):

```python
from pydantic import BaseModel, Field
from datetime import datetime

class TextProcessResponse(BaseModel):
    """Response from text processing."""
    
    original_text: str = Field(description="Original input text")
    processed_text: str = Field(description="Processed text")
    operation: str = Field(description="Operation performed")
    processing_time_ms: float = Field(description="Processing time in milliseconds")
    processed_at: datetime = Field(description="Processing timestamp")
```

### Step 2: Create Router

**Router Implementation** (`src/web_server/routers/text_router.py`):

```python
import time
from datetime import datetime
from fastapi import APIRouter, HTTPException
from src.web_server.models.request.text_request import TextProcessRequest
from src.web_server.models.response.text_response import TextProcessResponse

router = APIRouter(prefix="/api/text", tags=["text"])

def process_text(text: str, operation: str) -> str:
    """Process text based on operation."""
    if operation == "uppercase":
        return text.upper()
    elif operation == "lowercase":
        return text.lower()
    elif operation == "reverse":
        return text[::-1]
    elif operation == "word_count":
        return f"Word count: {len(text.split())}"
    else:
        raise ValueError(f"Unknown operation: {operation}")

@router.post("/process", response_model=TextProcessResponse)
async def process_text_endpoint(request: TextProcessRequest) -> TextProcessResponse:
    """Process text with specified operation."""
    try:
        start_time = time.time()
        
        processed_text = process_text(request.text, request.operation)
        
        processing_time = (time.time() - start_time) * 1000
        
        return TextProcessResponse(
            original_text=request.text,
            processed_text=processed_text,
            operation=request.operation,
            processing_time_ms=processing_time,
            processed_at=datetime.now()
        )
    except Exception as e:
        raise HTTPException(
            status_code=400,
            detail=f"Text processing failed: {str(e)}"
        )

@router.get("/operations")
async def get_operations():
    """Get list of available operations."""
    return {
        "operations": [
            "uppercase",
            "lowercase",
            "reverse",
            "word_count"
        ]
    }
```

## Example 3: Database Query Tool

An example showing how to create a tool that queries a database.

### Step 1: Define Models

**Input Model** (`src/mcp/models/input/database_input.py`):

```python
from pydantic import BaseModel, Field

class DatabaseQueryInput(BaseModel):
    """Input for database query tool."""
    
    query: str = Field(
        description="SQL query to execute",
        examples=["SELECT * FROM users LIMIT 10"]
    )
    limit: int = Field(
        default=100,
        ge=1,
        le=1000,
        description="Maximum number of rows to return"
    )
```

**Output Model** (`src/mcp/models/output/database_output.py`):

```python
from pydantic import BaseModel, Field
from typing import Optional

class DatabaseQueryOutput(BaseModel):
    """Output from database query tool."""
    
    rows: list[dict] = Field(description="Query results")
    row_count: int = Field(description="Number of rows returned")
    execution_time_ms: float = Field(description="Query execution time")
    error: Optional[str] = Field(
        default=None,
        description="Error message if query failed"
    )
```

### Step 2: Implement Tool

**Tool Implementation** (`src/mcp/tools/database.py`):

```python
import time
import sqlite3
from src.mcp import mcp_factory
from src.mcp.models.input.database_input import DatabaseQueryInput
from src.mcp.models.output.database_output import DatabaseQueryOutput

@mcp_factory.get().tool()
def query_database(input: DatabaseQueryInput) -> DatabaseQueryOutput:
    """
    Execute a database query.
    
    This example uses SQLite, but can be adapted for other databases.
    """
    try:
        start_time = time.time()
        
        # Connect to database
        conn = sqlite3.connect(':memory:')
        cursor = conn.cursor()
        
        # Execute query with limit
        query = f"{input.query} LIMIT {input.limit}"
        cursor.execute(query)
        
        # Fetch results
        columns = [description[0] for description in cursor.description]
        rows = [dict(zip(columns, row)) for row in cursor.fetchall()]
        
        execution_time = (time.time() - start_time) * 1000
        
        conn.close()
        
        return DatabaseQueryOutput(
            rows=rows,
            row_count=len(rows),
            execution_time_ms=execution_time
        )
    except Exception as e:
        return DatabaseQueryOutput(
            rows=[],
            row_count=0,
            execution_time_ms=0,
            error=f"Query failed: {str(e)}"
        )
```

### Extend Configuration

## Example 6: Error Handling and Validation

An example showing best practices for error handling and input validation.

### Input Validation

```python
from pydantic import BaseModel, Field, field_validator

class ValidatedInput(BaseModel):
    """Input with comprehensive validation."""
    
    email: str = Field(
        description="Email address",
        pattern=r"^[^@]+@[^@]+\.[^@]+$"
    )
    age: int = Field(
        ge=0,
        le=150,
        description="Age in years"
    )
    password: str = Field(
        min_length=8,
        description="Password (minimum 8 characters)"
    )
    
    @field_validator('password')
    @classmethod
    def validate_password(cls, v):
        """Validate password strength."""
        if not any(c.isupper() for c in v):
            raise ValueError('Password must contain uppercase letter')
        if not any(c.isdigit() for c in v):
            raise ValueError('Password must contain digit')
        return v
```

### Error Handling in Tools

```python
from typing import Optional

class SafeToolOutput(BaseModel):
    """Output with error handling."""
    
    success: bool = Field(description="Whether operation succeeded")
    data: Optional[dict] = Field(default=None, description="Operation result")
    error: Optional[str] = Field(default=None, description="Error message")
    error_code: Optional[str] = Field(default=None, description="Error code")

@mcp_factory.get().tool()
def safe_tool(input: ValidatedInput) -> SafeToolOutput:
    """Tool with comprehensive error handling."""
    try:
        # Validate input (Pydantic already does this)
        
        # Perform operation
        result = perform_operation(input)
        
        return SafeToolOutput(
            success=True,
            data=result
        )
    except ValueError as e:
        return SafeToolOutput(
            success=False,
            error=str(e),
            error_code="VALIDATION_ERROR"
        )
    except Exception as e:
        return SafeToolOutput(
            success=False,
            error=f"Unexpected error: {str(e)}",
            error_code="INTERNAL_ERROR"
        )
```
